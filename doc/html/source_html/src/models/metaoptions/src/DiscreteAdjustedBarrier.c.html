#include <math.h>
#include <assert.h>
#include <stdlib.h>

#include "metaoptions.h"

/* Discrete barrier monitoring adjustment */
double DiscreteAdjustedBarrier(double S, double H, double v, double dt) 
{
	double result;

    if (H > S) 
        result = H * exp(0.5826 * v * sqrt(dt));
	else if(H < S)
        result = H * exp(-0.5826 * v * sqrt(dt));
	else
		abort();

	return result;
}
