#include "defs.h"
#include <stdarg.h>

void logger(const char *name, const int num_args, ...)
{
  va_list ap;
  double dval;

  static FILE *fp = 0;
  static char logFile[512];
  char *homeDirectory;

  static int oneTime = 0;

  if( oneTime++ == 0 )
  {
    homeDirectory = getenv("HOME");

    if( homeDirectory != NULL )
       sprintf(logFile,"%s/%s.log", homeDirectory, PACKAGE);
     else
       sprintf(logFile,"%s.log", PACKAGE);
  }

  if( !fp && strcmp(name,"") )
  {
    fp = fopen(logFile,"w");

  } else if( fp && !strcmp(name,"") )
  {
    fprintf(fp,"logger closing debug file\n");
    fclose(fp);
    return;

  } else if( !fp && !strcmp(name,"") )
  {
    return;
  }

  //fprintf(fp,"num_args = %d\n",num_args);
  fprintf(fp,"%s\n", name);

  va_start(ap, num_args); /* make ap point to 1st unnamed arg */

  int index;
  for (index = 0; index < num_args; index++)
  {
    dval = (double) va_arg(ap, double);
    fprintf(fp,"%d = %.*f\n", index+1, 15, dval);
  }
  va_end(ap);

  fflush(fp);
}
